  • (BOOL)application:(UIApplication )application didFinishLaunchingWithOptions:(NSDictionary )launchOptions {
    // Override point for customization after application launch.

// MARK: 复习昨天的代码

/**
 *  设置window窗口
 */
self.window = [[UIWindow alloc] initWithFrame:[UIScreen mainScreen].bounds];
self.window.backgroundColor = [UIColor whiteColor];
[self.window makeKeyAndVisible];

/**
 *  设置根视图控制器
 */
self.window.rootViewController = [[ViewController alloc] init];

// !!!:创建nameLabel

UILabel *nameLabel = [[UILabel alloc] init];
// 为nameLabel设置frame属性
nameLabel.frame = CGRectMake(60, 80, 75, 40);
// 为nameLabel设置背景颜色属性
nameLabel.backgroundColor = [UIColor cyanColor];
// 为nameLabel设置文本属性
nameLabel.text = @"用户名";
// 为nameLabel设置文本格式属性
nameLabel.textAlignment = NSTextAlignmentCenter;
// 将nameLabel添加到window视图上
[self.window addSubview:nameLabel];

// !!!:创建pswLabel

UILabel *pswLabel = [[UILabel alloc] init];
// 为pswLabel设置frame属性
pswLabel.frame = CGRectMake(60, 140, 75, 40);
// 为pswLabel设置背景颜色属性
pswLabel.backgroundColor = [UIColor cyanColor];
// 为pswLabel设置文本属性
pswLabel.text = @"密码";
// 为pswLabel设置文本格式属性
pswLabel.textAlignment = NSTextAlignmentCenter;
// 将pswLabel添加到window视图上
[self.window addSubview:pswLabel];

// ???:今天开始学习的— 创建UITextField —

/**
 *  UITextField
 */
UITextField *nameTextField = [[UITextField alloc] init];

/**
 *  textField.frame
 *
 *  @param 150 textFieldX
 *  @param 80  textFieldY
 *  @param 200 textFieldW
 *  @param 40  textFieldH
 */
CGFloat textFieldX = 150;
CGFloat textFieldY = 80;
CGFloat textFieldW = 200;
CGFloat textFieldH = 40;
// 为nameTextField设置frame属性
nameTextField.frame = CGRectMake(textFieldX, textFieldY, textFieldW, textFieldH);
// 设置textField背景颜色
nameTextField.backgroundColor = [UIColor cyanColor];

// 第二步: 设置代理
nameTextField.delegate = self;

// 将textField添加到window视图上
[self.window addSubview:nameTextField];

/**
 *  UITextField
 */
UITextField *pswTextField = [[UITextField alloc] init];

/**
 *  pswTextField.frame
 *
 *  @param 150 textFieldX
 *  @param 80  textFieldY
 *  @param 200 textFieldW
 *  @param 40  textFieldH
 */
CGFloat pswTextFieldX = 150;
CGFloat pswTextFieldY = 140;
CGFloat pswTextFieldW = 200;
CGFloat pswTextFieldH = 40;
// 为pswTextField设置frame属性
pswTextField.frame = CGRectMake(pswTextFieldX, pswTextFieldY, pswTextFieldW, pswTextFieldH);
// 设置textField背景颜色
pswTextField.backgroundColor = [UIColor cyanColor];

// 第二步:设置代理
pswTextField.delegate = self;

// 将textField添加到window视图上
[self.window addSubview:pswTextField];

// MARK: 创建Button

UIButton *registerButton = [UIButton new];
// 为registerButton设置frame属性
registerButton.frame = CGRectMake(60, 200, 80, 40);
// 为registerButton设置背景颜色属性
registerButton.backgroundColor = [UIColor cyanColor];

//为button添加主题
[registerButton setTitle:@"注册" forState:UIControlStateNormal];
// 为button添加主题颜色
[registerButton setTitleColor:[UIColor blackColor] forState:UIControlStateNormal];
// 将button添加到window视图上
[self.window addSubview:registerButton];

/**
 *  此处创建UIButton是,使用便利构造器，如果类型使用的是system,那
 么加入前景图片时会填充,所以要使用custom.
 */
UIButton *cancelButton = [UIButton buttonWithType:UIButtonTypeCustom];
// 为cancelButton设置frame属性
cancelButton.frame = CGRectMake(160, 200, 80, 40);
// 为cancelButton设置背景颜色属性
cancelButton.backgroundColor = [UIColor cyanColor];
// 为cancelButton设置"前"景图片
[cancelButton setImage:[UIImage imageNamed:@"2.png"] forState:UIControlStateNormal];
// 为cancelButton设置"背"景图片
[cancelButton setBackgroundImage:[UIImage imageNamed:@"1"] forState:UIControlStateNormal];


// 为cancelButton添加主题
[cancelButton setTitle:@"取消" forState:UIControlStateNormal];
// 为cancelButton添加主题颜色
[cancelButton setTitleColor:[UIColor blackColor] forState:UIControlStateNormal];
// 将button添加到window视图上
[self.window addSubview:cancelButton];

// 创建登录button
UIButton *loginButton = [[UIButton alloc] init];
// 为登录button设置frame属性
loginButton.frame = CGRectMake(260, 200, 80, 40);
// 为登录button设置背景颜色属性
loginButton.backgroundColor = [UIColor cyanColor];


// 为loginButton添加主题
[loginButton setTitle:@"登录" forState:UIControlStateNormal];
// 为loginButton添加主题颜色方法
[loginButton setTitleColor:[UIColor blackColor] forState:UIControlStateNormal];
// 将button添加到window视图上
[self.window addSubview:loginButton];

// TODO: 程序的启动流程

return YES;

}
