nameTextField.backgroundColor = [UIColor cyanColor];

// MARK: 创建Button

// 为registerButton设置frame属性
registerButton.frame = CGRectMake(60, 200, 80, 40);
// 为registerButton设置背景颜色属性
registerButton.backgroundColor = [UIColor cyanColor];

// 为button添加主题颜色
[registerButton setTitleColor:[UIColor blackColor] forState:UIControlStateNormal];
// 将button添加到window视图上
[self.window addSubview:registerButton];

/**
 *  此处创建UIButton是,使用便利构造器，如果类型使用的是system,那
 么加入前景图片时会填充,所以要使用custom.
 */
UIButton *cancelButton = [UIButton buttonWithType:UIButtonTypeCustom];
// 为cancelButton设置frame属性
cancelButton.frame = CGRectMake(160, 200, 80, 40);
// 为cancelButton设置背景颜色属性
cancelButton.backgroundColor = [UIColor cyanColor];
// 为cancelButton设置"前"景图片
[cancelButton setImage:[UIImage imageNamed:@"2.png"] forState:UIControlStateNormal];
// 为cancelButton设置"背"景图片
[cancelButton setBackgroundImage:[UIImage imageNamed:@"1"] forState:UIControlStateNormal];


// 为cancelButton添加主题
[cancelButton setTitle:@"取消" forState:UIControlStateNormal];
// 为cancelButton添加主题颜色
[cancelButton setTitleColor:[UIColor blackColor] forState:UIControlStateNormal];
// 将button添加到window视图上
[self.window addSubview:cancelButton];

// 创建登录button
UIButton *loginButton = [[UIButton alloc] init];
// 为登录button设置frame属性
loginButton.frame = CGRectMake(260, 200, 80, 40);
// 为登录button设置背景颜色属性
loginButton.backgroundColor = [UIColor cyanColor];


// 为loginButton添加主题
[loginButton setTitle:@"登录" forState:UIControlStateNormal];
// 为loginButton添加主题颜色方法
[loginButton setTitleColor:[UIColor blackColor] forState:UIControlStateNormal];
// 将button添加到window视图上
[self.window addSubview:loginButton];

  • (BOOL)application:(UIApplication )application didFinishLaunchingWithOptions:(NSDictionary )launchOptions {
    // Override point for customization after application launch.

    // 应用程序启动后,将需要写的代码写在此处

#pragma mark - UIWindow -

😄

/**
 *  UIWindow是一种特殊的UIView，通常在一个app中只会有一个UIWindow
 *  iOS程序启动完毕后，创建的第一个视图控件就是UIWindow
 *  一个iOS程序之所以能显示到屏幕上，完全是因为它有UIWindow。也就说，
 *  没有UIWindow，就看不见任何UI界面
 */
self.window = [[UIWindow alloc] initWithFrame:[UIScreen mainScreen].bounds];

// 为窗口设置背景颜色
self.window.backgroundColor = [UIColor whiteColor];

// 让窗口成为主窗口，并且显示出来。有这个方法，才能把信息显示到屏幕上。
[self.window makeKeyAndVisible];

// 设置根视图控制器,自xcode 7之后,创建window之后,就一定要设定根视图控制器,否则会崩溃. 写上此句代码之后就没有问题了.
self.window.rootViewController = [[ViewController alloc] init];

此时运行程序的效果是这个样子的

#pragma mark - 属性:frame -

// 创建一个view对象
UIView *view = [[UIView alloc] init];
/**
 *  frame: 控制位置和大小.
 *
 *  @param (位置)x--负责控制手机屏幕的横向坐标frame.origin.x
 *  @param (位置)y--负责控制手机屏幕的纵向坐标frame.origin.y
 *  @param width--负责控制控件在屏幕上显示的宽度 frame.size.width
 *  @param height--负责控件控件在屏幕上显示的高度frame.size.height
 */
// view.frame = CGRectMake(<#CGFloat x#>, <#CGFloat y#>, <#CGFloat width#>, <#CGFloat height#>)

/**
 *  为view对象设置frame的大小
 *
 *  @param viewX 为窗口的x大小
 *  @param viewY 为窗口的y大小
 *  @param viewW 为窗口的w大小
 *  @param viewH 为窗口的h大小
 */
CGFloat viewW = 200;
CGFloat viewH = 200;
CGFloat viewX = 110;
CGFloat viewY = 110;

view.frame = CGRectMake(viewX, viewY, viewW, viewH);

// 为view设置颜色
view.backgroundColor = [UIColor blackColor];

// 将view添加到窗口视图上
[self.window addSubview:view];

此时运行程序的效果是这个样子的

黑色部分是UIView

#pragma mark - 属性:bounds -

/**
 *  frame指的是：该view在父view坐标系统中的位置和大小。（参照点是父亲的坐标系统）

 *  bounds指的是：该view在本身坐标系统中的位置和大小。（参照点是本身坐标系统）
 */
NSLog(@"%@",NSStringFromCGRect(view.frame));
NSLog(@"%@",NSStringFromCGRect(view.bounds));
// 如果你还是不明白,那可以参照图片frame&bounds

#pragma mark - 属性:center -

/**
 *  center:控制位置的中心点
 */

NSLog(@"中心点:%@",NSStringFromCGPoint(view.center));

// center的内部实现
CGPoint myCenter = CGPointMake(view.bounds.size.width/2 + view.frame.origin.x, view.bounds.size.height / 2 + view.frame.origin.y);
NSLog(@"%f,%f",myCenter.x,myCenter.y);

return YES;

}
